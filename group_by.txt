1. Contare quanti iscritti ci sono stati ogni anno
SELECT COUNT(`id`) AS `n_iscritti`, YEAR(`enrolment_date`) AS `anno_iscrizione`
FROM `students`
GROUP BY `anno_iscrizione`;
